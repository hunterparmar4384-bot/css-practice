#include <iostream>
using namespace std;
/*
virtual -> memory override
*/

class one
{
  public:
  int a = 10;
  one()
  {
      cout<<"hello one"<<endl;
  }
};

class two : virtual public one
{
  public:
  two()
  {
      cout<<"hello two"<<endl;
  }
};

class tr : virtual public one
{
  public:
  tr()
  {
      cout<<"hello tr"<<endl;
  }
};

class four :public two, public tr
{
  public:
  four()
  {
      cout<<"hello four"<<endl;
      cout<<"a = "<<a<<endl;
  }
};


int main() {
    four p;
    return 0;
}

    {
        cout<<"class B"<<endl;
    }

{
  public:
     C()
    {
        cout<<"class c"<<endl;

***** single inheritance *****
class parent 
{
    public:
    int a = 10;
    parent()
    {
        cout<<"parent class\n";
    }
};
class child : public parent
{
    public:
    child()
    {
        cout<<"child class\n";
    }
};
int main() {
    child c;
    cout<<"c.a = "<<c.a<<endl;
    return 0;
}
